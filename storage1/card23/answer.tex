    \ddot{\theta} + \frac{g}{\ell }\theta = 0

    \item $\displaystyle \omega = \sqrt{\frac{g}{\ell}}$ 
    \item $\displaystyle \omega = \sqrt{\frac{mgR}{I}}$ ($R$ is the distance from the center of mass to the pivot)
\end{enumerate}
