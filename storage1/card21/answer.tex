\begin{itemize}
    \item $m \ddot{x} + b \dot{x} + k x = 0$ ($\displaystyle \beta = \frac{b}{2m}$, $\displaystle \omega_0 = \sqrt{\frac{k}{m}}$) 

    \item Let $\omega^2 = \beta^2 - \omega_0^2$ 
    \begin{enumerate}
        \item underdamped: $\omega^2 < 0$
        \item critically damped: $\omega^2 = 0$
        \item overdamped: $\omega^2 > 0$
    \end{{enumerate}
\end{itemize}
